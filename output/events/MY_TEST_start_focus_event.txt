country_event = {# start focus event
	id = TAN_fritz_todt_focus_control.01
	
	hidden = yes

	is_triggered_only = yes

	trigger = {
		MY_TEST_TAN_fritz_todt_is_on_the_seat = yes
		NOT = {has_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag}
	}

	immediate = {
		set_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag
				if = {
			limit = {

OR = { has_completed_focus = TAN_fritz_todt_enable
			 has_completed_focus = TAN_albert_speer_enable
			}
			
				NOT = { has_shine_effect_on_focus = TAN_industry_development}
				NOT = { has_completed_focus = TAN_industry_development }
                NOT = { has_country_flag = MY_TEST_TAN_industry_development_waiting }
			}
			activate_shine_on_focus = TAN_industry_development
			set_country_flag = { flag = MY_TEST_TAN_industry_development_in_progress_flag days = 15 value = 1 }
			set_variable = { MY_TEST_TAN_industry_development_doing_focus_advisor = 1 }
			country_event = { id = TAN_fritz_todt_focus_control.02 days = 15 }

		}
		else_if = {
			limit = {

OR = { has_completed_focus = TAN_fritz_todt_enable
			}
			
				NOT = { has_shine_effect_on_focus = TAN_fritz_todt_upgrade}
				NOT = { has_completed_focus = TAN_fritz_todt_upgrade }
                NOT = { has_country_flag = MY_TEST_TAN_fritz_todt_upgrade_waiting }
			}
			activate_shine_on_focus = TAN_fritz_todt_upgrade
			set_country_flag = { flag = MY_TEST_TAN_fritz_todt_upgrade_in_progress_flag days = 15 value = 1 }
			set_variable = { MY_TEST_TAN_fritz_todt_upgrade_doing_focus_advisor = 1 }
			country_event = { id = TAN_fritz_todt_focus_control.02 days = 15 }

		}

				else_if = {
			limit = {
				MY_TEST_TAN_fritz_todt_all_focus_completed = yes
			}
			#log = "TAN_fritz_todt completed all his focus"
		}


				else = { #Debug logs
			#log = "SOMETHING WENT WRONG WITH TAN_fritz_todt's FOCUS BRANCH - Cannot start new focus"
			set_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag
			country_event = { id = TAN_fritz_todt_focus_control.01 days = 1 random_days = 1 }
		}

	}
}
