country_event = {
	id = TAN_fritz_todt_focus_control.02

	hidden = yes

	is_triggered_only = yes

	trigger = {
		MY_TEST_TAN_fritz_todt_is_on_the_seat = yes
	}

	immediate = {
				if = {
			limit = {
				has_shine_effect_on_focus = TAN_industry_development
				check_variable = { MY_TEST_TAN_industry_development_doing_focus_advisor = 1 } 
				
OR = { has_completed_focus = TAN_fritz_todt_enable
			 has_completed_focus = TAN_albert_speer_enable
			}
			
			}
			complete_national_focus = {
				focus = TAN_industry_development
				use_side_message = yes
				originator_name = TAN_fritz_todt
			}
			clr_country_flag = MY_TEST_TAN_industry_development_in_progress_flag
			clr_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag
			country_event = { id = TAN_fritz_todt_focus_control.01 days = 1 random_days = 1 }
		}
		else_if = {
			limit = {
				has_shine_effect_on_focus = TAN_fritz_todt_upgrade
				check_variable = { MY_TEST_TAN_fritz_todt_upgrade_doing_focus_advisor = 1 } 
				
OR = { has_completed_focus = TAN_fritz_todt_enable
			}
			
			}
			complete_national_focus = {
				focus = TAN_fritz_todt_upgrade
				use_side_message = yes
				originator_name = TAN_fritz_todt
			}
			clr_country_flag = MY_TEST_TAN_fritz_todt_upgrade_in_progress_flag
			clr_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag
			country_event = { id = TAN_fritz_todt_focus_control.01 days = 1 random_days = 1 }
		}

				else = { #Debug logs
			#log = "SOMETHING WENT WRONG WITH TAN_fritz_todt's FOCUS BRANCH - Cannot complete focus"
			clr_country_flag = MY_TEST_TAN_fritz_todt_upgrade_in_progress_flag
			clr_country_flag = TAN_fritz_todt_has_start_focus_control_event_flag
			country_event = { id = TAN_fritz_todt_focus_control.01 days = 1 random_days = 1 }

		}

	}
}
